import favorite from '../controller/favorites';
/**
 *
 * @param {req} req
 * @param {res} res
 * @param {next} next
 * @return { message } message
 */
export default function (req, res, next) {
  const { recipeId, userId } = req.body;
  if (favorite) {
    if (!recipeId || recipeId.trim() === '') {
      return res.status(400).send({
        message: 'Please enter a valid recipeid!'
      });
    }
 
    if (!userId || userId.trim() === '') {
      return res.status(400).send({
        message: 'Please enter a valid userid!'
      });
    }
  } else {
    return res.status(400).send({ message: 'All fields must be provided!' });
  }
 
 
  favorite.findAll({
    where: {
      recipeId,
      userId
    },
  })
    .then((createdFavorite) => {
      if (createdFavorite.length >= 1) {
        return res.status(200).send({
          message: 'You have already favorited this recipe'
        });
      }
    })
    .catch((err) => {
      res.status(500).send({ err });
    });
 
  next();
}
